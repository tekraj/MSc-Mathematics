	\subsection{$2y(a-x) dx + [z-y^2 + (a-x)^2] dy - y dz = 0$}
	\text{We are given the PfDE}
	\begin{equation}
		2y(a-x) dx + [z-y^2 + (a-x)^2] dy - y dz = 0
		\label{eq:exact11}
	\end{equation}
	\text{Let's define the vector $\vec{X} = (P\,,Q\, ,R\,)$, where ${P\,=\,2y(a-x)}, {Q\,=z-y^2+(a-x)^2,\, R = -y}$}
	\text{so ${\vec{X}=((2y(a-x)),(z-y^2+(a-x)^2),-y)}$}.\\
	To check the integrability, lets calculate the ${\vec{X} \cdot (\nabla \times \vec{X})}$
	
	\begin{equation}
		\begin{split}
			\vec{X} \cdot (\nabla \times \vec{X}) &= 
			\begin{vmatrix}
				P & Q & R \\
				\frac{\partial}{\partial x} &  \frac{\partial}{\partial y} &  \frac{\partial}{\partial z} \\
				P & Q & R \\
			\end{vmatrix} \\
			\vec{X} \cdot (\nabla \times \vec{X}) &= 
			\begin{vmatrix}
				2y(a-x) & (z-y^2+(a-x)^2 & -y \\
				\frac{\partial}{\partial x} &  \frac{\partial}{\partial y} &  \frac{\partial}{\partial z} \\
				2y(a-x) & (z-y^2+(a-x)^2 & -y \\
			\end{vmatrix} \\
			%
			\vec{X} \cdot (\nabla \times \vec{X}) &= 2y(a-x)(-1-1) + (z-y^2+(a-x)^2)(0-0) - y(-2(a-x) - 2a+2x) \\
			\vec{X} \cdot (\nabla \times \vec{X}) &= -4y(a-x)-y(-2a+2x-2a+2x) \\
			\vec{X} \cdot (\nabla \times \vec{X})	&= -4y(a-x) - y (-4a + 4x) \\
			%
			\vec{X} \cdot (\nabla \times \vec{X})	&= -4ay + 4xy + 4ay - 4xy \\
			%
			\vec{X} \cdot (\nabla \times \vec{X}) & = 0
		\end{split}
	\end{equation}
	\text{Since $\vec{X} \cdot \operatorname{curl} \vec{X} = 0$, the given Pfaffian Differential Equation~\eqref{eq:exact11} is integrable.}\\
	\text{Dividing the given PfDE by $y^2$ on both sides}
	\begin{equation}
		\begin{split}
			&\frac{2}{y} (a-x)\, dx + \frac{z}{y^2} \,dy - dy + \frac{(a-x)^2}{y^2}\, dy - \frac{1}{y} \,dz &= 0 \\
			&\frac{2}{y} (a-x)\, dx +  \frac{(a-x)^2}{y^2} dy + \frac{z}{y^2} dy - dy - \frac{1}{y} dz &= 0  
		\end{split}
		\label{eq:exact12}
	\end{equation}
	

	\text{Here the terms $\frac{2}{y}(a - x)\,dx + \frac{(a - x)^2}{y^2}\,dy
	$ can be written as the differential of $-\frac{(a-x)^2}{y}$ }
	\begin{align*}
		d(-\frac{(a-x)^2}{y}) = - d(\frac{(a-x)^2}{y}) = \frac{2}{y}(a - x)\,dx + \frac{(a - x)^2}{y^2}\,dy
	\end{align*}
	\text{Also the remaining terms $\frac{z}{y^2} dy - \frac{1}{y} dz - dy$ can be written as the differential of (-$\frac{z}{y} - y$)}
	\begin{align*}
		d(-\frac{z}{y} - y) = 	- (\frac{z}{y^2} dy + \frac{1}{y} dz) - dy = \frac{z}{y^2} dy - \frac{1}{y}dz - dy
	\end{align*}
	\text{Thus the equation ~\eqref{eq:exact12} becomes}\\
	\[
	d(-\frac{(a-x)^2}{y}) +d(-\frac{z}{y} - z) = 0
	\]\\
	\text{Combining the differentials:}\\
	\begin{align*}
		&d(-\frac{(a-x)^2}{y} - \frac{z}{y} - y) = 0\\
		&d(\frac{(a-x)^2}{y} + \frac{z}{y} + y) = 0\\
	\end{align*}
	\text{Integrating on both sides}\\
	\[
	frac{(a-x)^2}{y} +\frac{z}{y} + y = C, where C is an arbitrary constant.
	\]
	This is the required primitives of the given PfDE.

	\subsection{$y(1+z^2) dx - x(1+z^2) dy + (x^2+y^2) dz = 0$}
	\setcounter{equation}{0}
	\text{We are given the PfDE}
	\begin{equation}
		y(1+z^2) dx - x(1+z^2) dy + (x^2+y^2) dz = 0
		\label{eq:exact21}
	\end{equation}
	\text{Let's define the vector $\vec{X} = (P\,,Q\, ,R\,)$, where $P\,=\,y(1+z^2), Q\,= -x(1+z^2),\, R = x^2+y^2$}
	\text{so $\vec{X}=(y(1+z^2),\,x(1+z^2), x^2+y^2)$}.\\
	To check the integrability, lets calculate the ${\vec{X} \cdot (\nabla \times \vec{X})}$
	
	\begin{align*}
			\vec{X} \cdot (\nabla \times \vec{X}) &= 
			\begin{vmatrix}
				P & Q & R \\
				\frac{\partial}{\partial x} &  \frac{\partial}{\partial y} &  \frac{\partial}{\partial z} \\
				P & Q & R \\
			\end{vmatrix} \\
			%
			\vec{X} \cdot (\nabla \times \vec{X}) &= 
			\begin{vmatrix}
				y(1+z^2) & -x(1+z^2) &  x^2+y^2 \\
				\frac{\partial}{\partial x} &  \frac{\partial}{\partial y} &  \frac{\partial}{\partial z} \\
				y(1+z^2) & -x(1+z^2) &  x^2+y^2 \\
			\end{vmatrix} \\
			%
			\vec{X} \cdot (\nabla \times \vec{X}) &= y(1+z^2) (2y+2xz) + x(1+z^2) (2x - 2yz) + (x^2+y^2) (-1-z^2  - 1-z^2 ) \\
			\vec{X} \cdot (\nabla \times \vec{X}) &= (1+z^2)(2y^2 - 2xyz+ 2x^2  + 2xyz) -2(x^2+y^2)(1-z^2)  \\
			\vec{X} \cdot (\nabla \times \vec{X})	&= (1+z^2)(2y^2 + 2x^2 - 2x^2 - 2y^2 ) \\
			%
			\vec{X} \cdot (\nabla \times \vec{X})	&=  0\\
	\end{align*}
	\text{Since $\vec{X} \cdot \operatorname{curl} \vec{X} = 0$, the given Pfaffian Differential Equation~\eqref{eq:exact21} is integrable.}\\
	\\
	\text{Dividing the given PfDE by $(x^2+y^2) (1+z^2)$}, we get \\
	\[\frac{y}{x^2+y^2} dx - \frac{x}{x^2+y^2} dy + \frac{1}{1+z^2} dz = 0\]\\
	\text{The terms $\frac{y}{x^2+y^2} - \frac{x}{x^2+y^2}$  can be expressed as the differential  $d(-\tan^{-1} \frac{y}{x})$}\\
	\[-d(\tan^{-1} \frac{y}{x}) = - (\frac{xdy-ydx}{x^2+y^2})\]\\
	\text{Also $d(\tan^{-1} z) = \frac{dz}{1+z^2}$}\\
	\text{Hence the equation becomes}\\
	\[-d(\tan{-1} \frac{y}{x}) + d(\tan^{-1} z) = 0\]\\
	\text{Integrating both sides we get}\\
	\[\tan^{-1} z - \tan^{-1} \frac{y}{x} = C , \text{where C is an arbitrary constant} \]
	\text{This is the require primitives}

	\subsection{$(y^2+yz+z^2) dx + (z^2+zx+x^2) dy + (x^2+xy+y^2) dz = 0$}
	\setcounter{equation}{0}
	\text{We are given the PfDE}
	\begin{equation}
			(y^2+yz+z^2) dx + (z^2+zx+x^2) dy + (x^2+xy+y^2) dz = 0
		\label{eq:exact31}
	\end{equation}
	Let's define the vector $\vec{X} = (P\,,Q\, ,R\,)$, where $P\,=\,(y^2+yz+z^2)$\\
	$Q\,= (z^2+zx+x^2),\, R = (x^2+xy+y^2)$\\
	\text{So $\vec{X}=((y^2+yz+z^2),\,(z^2+zx+x^2), (x^2+xy+y^2))$}.\\
	To check the integrability, lets calculate the ${\vec{X} \cdot (\nabla \times \vec{X})}$
	
	\begin{align*}
		\vec{X} \cdot (\nabla \times \vec{X}) &= 
		\begin{vmatrix}
			P & Q & R \\
			\frac{\partial}{\partial x} &  \frac{\partial}{\partial y} &  \frac{\partial}{\partial z} \\
			P & Q & R \\
		\end{vmatrix} \\
		%
		\vec{X} \cdot (\nabla \times \vec{X}) &= 
		\begin{vmatrix}
			(y^2+yz+z^2) & (z^2+zx+x^2) &  (x^2+xy+y^2) \\
			\frac{\partial}{\partial x} &  \frac{\partial}{\partial y} &  \frac{\partial}{\partial z} \\
				(y^2+yz+z^2) & (z^2+zx+x^2) &  (x^2+xy+y^2) \\
		\end{vmatrix} \\
		%
		\vec{X} \cdot (\nabla \times \vec{X}) &= (y^2+yz+z^2)(2y-2z) + (z^2+zx+x^2)(2z-2x) \\
			\;&\quad{} + (x^2+xy+y^2)(2x-2y) \\
		\vec{X} \cdot (\nabla \times \vec{X}) & = 2y^3 + 2y^2z + 2yz^2 - 2y^2z - 2yz^2 - 2z^3 \\
		\;&\quad{}+2z^3 + 2z^2x + 2zx^2 - 2zx^2 - 2x^3 - 2zx^2 \\
		\;&\quad{}+2x^3 + 2x^2y + 2xy^2 - 2x^2y - 2xy^2 - 2y^3 \\
		%
		\vec{X} \cdot (\nabla \times \vec{X})	&=  0\\
	\end{align*}
	\text{Since $\vec{X} \cdot \operatorname{curl} \vec{X} = 0$, the given Pfaffian Differential Equation~\eqref{eq:exact21} is integrable.}\\
	\\
	\text{Since the given PfDE is homogeneous of degree 2}\\
	\text{Let $y= ux$ and $z = vx$} \\
	\[
	 dy = udx + x du \\
	 dz = vdx + x dv
	\]
	\text{Then the given PfDE becomes}
	\begin{align*}
		&x^2(u^2 + uv + v^2)\,dx
		+ x^2u\,(v^2+v+1)\,dx
		+ x^2v\,(u^2+u+1)\,dx\\
		&\quad{}+\,x^3(v^2+v+1)\,du
		+\,x^3(u^2+u+1)\,dv
		=0\\
		\quad
		&x^2\Bigl[(u^2+uv+v^2)+u(v^2+v+1)+v(u^2+u+1)\Bigr]\,dx\\
		&\quad{}+x^3\,(v^2+v+1)\,du
		+x^3\,(u^2+u+1)\,dv
		=0.\\
		\quad
		&x^2\Bigl[(u+v+1)(u+uv+v) \Bigl] dx\\
		&\quad{}+x^3\,(v^2+v+1)\,du	\\
		&\quad{}+x^3(u^2+u+1)dv	=0.
	\end{align*}
	\text{Dividing both sides by $\frac{1}{x^3\Bigl[(u+v+1)(u+uv+v)\Bigl]}$}
	\begin{align*}
		\frac{1}{x} dx + \frac{v^2+v+1}{(u+v+1)(u+uv+v)} du + \frac{u^2+u+1}{(u+v+1)(u+uv+v)} dv = 0 
	\end{align*}
	\text{Using Partial Fraction}\\
	\begin{align*}
		\frac{v^2+v+1}{(u+v+1)(u+uv+v)} &= \frac{v+1}{u+uv+v} - \frac{1}{u+v+1} \\
		\frac{u^2+u+1}{(u+v+1)(u+uv+v)} &= \frac{u+1}{u+uv+v} - \frac{1}{u+v+1}\\
	\end{align*}
	\text{So,}\\
	\begin{align*}
		\frac{1}{x} dx + \frac{v+1}{u+uv+v} du - \frac{1}{u+uv+1}du + \frac{u+1}{u+uv+v} dv - \frac{1}{u+v+1} dv = 0
	\end{align*}
	\text{Integrating we get}
	\begin{flushleft}
		\begin{align*}
			&\ln(x) + \ln(u + uv + v) - \ln(u + v + 1) + \ln(u + uv + v) - \ln(u + v + 1) = C, \\
			&\text{where \( C \) is an arbitrary constant} \\
			&\ln\left(x\left(\frac{u + uv + v}{u + v + 1}\right)^2\right) = C \\
			&\text{Exponentiating both sides:} \\
			&x\left(\frac{u + uv + v}{u + v + 1}\right)^2 = e^C = C_1 \\
			&\text{Substituting } u = \frac{y}{x},\ v = \frac{z}{x}: \\
			&x\left( \frac{ \frac{y}{x} + \frac{y}{x} \cdot \frac{z}{x} + \frac{z}{x} }{ \frac{y}{x} + \frac{z}{x} + 1 } \right)^2 = C_1 \\
			&= x\left( \frac{ \frac{y + z}{x} + \frac{yz}{x^2} }{ \frac{y + z + x}{x} } \right)^2 \\
			&= x\left( \frac{ y + z + \frac{yz}{x} }{ y + z + x } \cdot \frac{1}{x} \right)^2 \\
			&= x\left( \frac{ yx + zx + yz }{ x(y + z + x) } \right)^2 \\
			&= x \cdot \left( \frac{ yx + yz + zx }{ x(y + z + x) } \right)^2 \\
			&= \frac{(yx + yz + zx)^2}{x(y + z + x)^2} = C_1 \\
			&\Rightarrow \left( \frac{yx + yz + zx}{y + z + x} \right)^2 = C_1 x \\
		\end{align*}
	\end{flushleft}
	
	\text{Hence, the required primitive is:}
	
	\[
	\boxed{ \big(\frac{yx + yz + zx}{y + z + x}\big)^2 = C_1x }
	\]
	
	
	% ============  4 ======================%
	\subsection{$yz\,dx + xz\,dy + xy\,dz = 0$}
	\setcounter{equation}{0}
	
	We are given the Pfaffian differential equation:
	\begin{equation}
		yz\,dx + xz\,dy + xy\,dz = 0
		\label{eq:exact41}
	\end{equation}
	
	Define the vector  \( \vec{X} = (P, Q, R) \), where:
	\[
	P = yz, \quad Q = xz, \quad R = xy
	\]
	
	Let's compute \( \vec{X} \cdot (\nabla \times \vec{X}) \):
	
	\[
	\vec{X} \cdot (\nabla \times \vec{X}) = 
	\begin{vmatrix}
		P & Q & R \\
		\frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\
		P & Q & R
	\end{vmatrix}
	\]
	
	Expanding the determinant:
	
	\begin{align*}
	\vec{X} \cdot (\nabla \times \vec{X}) =  
	yz \left( \frac{\partial}{\partial y}(xy) - \frac{\partial}{\partial z}(xz) \right)
	- xz \left( \frac{\partial}{\partial x}(xy) - \frac{\partial}{\partial z}(yz) \right)
	+ xy \left( \frac{\partial}{\partial x}(xz) - \frac{\partial}{\partial y}(yz) \right)
	\end{align*}
	\[
	\vec{X} \cdot (\nabla \times \vec{X}) =  
	yz (x - x) - xz (y - y) + xy (z - z)
	= 0
	\]
	
	Since \( \vec{X} \cdot (\nabla \times \vec{X}) = 0 \), the given Pfaffian differential equation is \textbf{integrable}.
	
	\vspace{1em}
	
	The left-hand side expression can be written as:
	\[
	yz\,dx + xz\,dy + xy\,dz = d(xyz)
	\]
	
	Hence, the equation becomes:
	\[
	d(xyz) = 0
	\]
	
	Integrating both sides:
	\[
	xyz = C, \quad \text{where \( C \) is an arbitrary constant.}
	\]
	
	\textbf{This is the general solution of the given PfDE.}
	 
	 % ============  5 ======================%
	 \subsection{$ (1+yz) dx + x (z-x) dy - (1+xy) dz = 0$}
	 \setcounter{equation}{0}
	 We are given the Pfaffian differential equation:
	 \begin{equation}
	 	(1 + yz)\,dx + x(z - x)\,dy - (1 + xy)\,dz = 0
	 	\label{eq:exact51}
	 \end{equation}
	 
	 Lets define the vector \( \vec{X} = (P, Q, R) \), where:
	 \[
	 P = 1 + yz, \quad Q = x(z - x), \quad R = -(1 + xy)
	 \]
	 
	 Lets check integrability by computing $\vec{X} \cdot (\nabla \times \vec{X})$
		\[
		\vec{X} \cdot (\nabla \times \vec{X}) = 
		\begin{vmatrix}
			P & Q & R \\
			\frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\
			P & Q & R
		\end{vmatrix}
		=
		\begin{vmatrix}
			1 + yz & x(z - x) & -(1 + xy) \\
			\frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\
			1 + yz & x(z - x) & -(1 + xy)
		\end{vmatrix}
		\]
		
		\begin{align*}
			&= (1 + yz)(-x - x) - x(z - x)(-y - y) - (1 + xy)(z - 2x - z) \\
			&= -2x(1 + yz) + 2xy(z - x) + 2x(1 + xy) \\
			&= (-2x - 2xyz) + 2xyz - 2x^2y + 2x + 2x^2y \\
			&\vec{X} \cdot (\nabla \times \vec{X}) = 0
		\end{align*}
		
		 Since \( \vec{X} \cdot (\nabla \times \vec{X}) = 0 \), the Pfaffian differential equation is \textbf{integrable}.\\
		 
		Let \( y = \text{constant} \), \(\Rightarrow\) \( dy = 0 \)
		
		Then the given PfDE reduces to:
		\begin{align*}
			(1 + yz) \, dx - (1 + xy) \, dz &= 0 \\
			\Rightarrow \frac{dx}{1 + xy} &= \frac{dz}{1 + yz}
		\end{align*}
		Integrating while treating \( y \) as constant:
		\begin{align*}
			\int \frac{dx}{1 + xy} &= \int \frac{dz}{1 + yz} \\
			\Rightarrow \frac{1}{y} \ln(1 + xy) &= \frac{1}{y} \ln(1 + yz) + C 
		\end{align*}
		\begin{equation}
			\begin{aligned}
				\Rightarrow \frac{1 + xy}{1 + yz} &= \phi(y)
			\end{aligned}
			\label{eq:exact52}
		\end{equation}
		
		
		Now set \( z = \alpha = 1 \), hence \( dz = 0 \)
		
		So the original PfDE becomes:
		\begin{align*}
			(1 + y) \, dx + x(1 - x) \, dy &= 0 \\
			\Rightarrow \frac{dx}{x(1 - x)} &= -\frac{dy}{1 + y}
		\end{align*}
		\text{Integrating on both sides, we get}
		\begin{align*}
			\ln(x) - \ln(1-x) &= -\ln(1+y) + C_2 \\
			\Rightarrow \ln\left(\frac{x(1+y)}{(1-x)}\right) &= C_2 \\
			\text{Exponentially both sides} \\
		\end{align*}
		\begin{equation}
			\frac{x(1+y)}{1-x} = C, \quad \text{where} \quad C = e^{C_2} \\			
		\end{equation}
		\text{Substituting } z = 1 \text{ in equation (2)} \\
		\begin{equation}
			\frac{1+xy}{1+y} = \phi(y)\\
		\end{equation}
		\begin{align*}
			\intertext{Also from equation (3)} 
			x(1+y)&=C - Cx \\
			\Rightarrow x&=\frac{C}{1 + y + C} \\
			\intertext{Using this value of \(x\) in equation (4), we get}
			\frac{1}{1+y} + \frac{C y}{1 + y + C}&=\phi(y) \\
			\Phi(y)&=\frac{1 + C}{1 + y + C}
		\end{align*}
		
		\text{Substituting $\phi(y) in equation ~\eqref{eq:exact52}$ we get}\\
		\\
		$\frac{1+xy}{1+yz} = \frac{1+C}{1+y+C}$\\
		\\
		\text{This is the required general solution of given PfDE}
		
		
		 % ============  6 ======================%
		\subsection{$ y(x+4) (y+z) dx - x(y+3z) dy + 2xy dz = 0$}
		\setcounter{equation}{0}
		We are given the Pfaffian differential equation:
		\begin{equation}
			y(x+4) (y+z) \, dx - x(y+3z) \, dy + 2xy \, dz = 0
			\label{eq:exact61}
		\end{equation}
		
		Let us define the vector \( \vec{X} = (P, Q, R) \), where:
		\[
		P = y(x+4)(y+z), \quad Q = -x(y+3z), \quad R = 2xy
		\]
		
		Let us check integrability by computing \( \vec{X} \cdot (\nabla \times \vec{X}) \):
		\[
		\vec{X} \cdot (\nabla \times \vec{X}) = 
		\begin{vmatrix}
			P & Q & R \\
			\frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\
			P & Q & R
		\end{vmatrix}
		=
		\begin{vmatrix}
			y(x+4)(y+z) & -x(y+3z) & 2xy \\
			\frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\
			y(x+4)(y+z) & -x(y+3z) & 2xy
		\end{vmatrix}
		\]
		
		\begin{align*}
			&= y(x+4)(y+z)( 2x +3x)
			- (-x(y+3z)) \left( 2y - y(x+4) \right) \\
			&\quad + 2xy \left( - (y+3z) - (x+4)(2y+z) \right) \\
			&= 5xy(x+4)(y+z) - xy (y+3z) (x+2) + 2xy(-9y - 7z - 2xy - xz) \\
			&= xy\big[ 5xy + 5xz + 20y + 20z -xy - 2y - 3xz - 6z - 18y - 14z - 4xy - 2xz\big] \\
			& \vec{X} \cdot (\nabla \times \vec{X}) = 0
		\end{align*}
		
		Since \( \vec{X} \cdot (\nabla \times \vec{X}) = 0 \), the Pfaffian differential equation is \textbf{integrable}.\\
		\text{Let x = constants this implies dx = 0}\\
		\text{Then the given PfDE reduced to }\\
		\begin{align*}
			-\,x(y+3z)\,dy + 2xy\,dz &= 0 \\
			2xy\,dz &= x(y+3z)\,dy \\
			\frac{dz}{dy} &= \frac{y+3z}{2y} \\
			\frac{dz}{dy} - \frac{3z}{2y} &= \frac12
		\end{align*}
		\noindent
		This is linear in \(z\), and the integrating factor is
		\[
		\mu(y)
		= \exp\!\Bigl(-\!\int \frac{3}{2y}\,dy\Bigr)
		= \exp\!\bigl(-\tfrac32\ln y\bigr)
		= y^{-\tfrac32}.
		\]
		\noindent Multiplying the above ODE by \(\mu\):
		\begin{align*}
			y^{-\tfrac32}\,\frac{dz}{dy}
			- \frac{3}{2y}\,y^{-\tfrac32}\,z
			&= \tfrac12\,y^{-\tfrac32} \\
			\frac{d}{dy}\bigl(z\,y^{-\tfrac32}\bigr)
			&= \tfrac12\,y^{-\tfrac32} \\
			\text{Integrating both sides:} \notag\\
			z\,y^{-\tfrac32}
			&= \tfrac12\int y^{-\tfrac32}\,dy + C(x) \notag\\
			&= \tfrac12\!\bigl(\tfrac{y^{-\tfrac12}}{-\tfrac12}\bigr) + \phi(x) \notag\\
		\end{align*}
		\begin{equation}
			\frac{z + y}{y^{-\tfrac32}}	= \phi(x)
			\label{eq:exact62}
		\end{equation}
		\text{Now set $ y=\alpha = 1$ this gives $ dy = 0 $}\\
		\noindent Then the given PfDE becomes
		\begin{align*}
			(x+4)(1+z)\,dx + 2x\,dz &= 0\\
			\frac{x+4}{2x}\,dx &= -\frac{dz}{1+z}\\
			\Bigl(\tfrac12 + \frac{2}{x}\Bigr)\,dx &= -\frac{dz}{1+z}\\
			\text{Integrating we get}\\
			\tfrac12\,x + 2\ln(x) &= -\ln(1+z) + C_1,\quad C_1\text{ is an arbitrary constant}\\
			\text{Exponentiating both sides}\\
			x^2 (1+z) &= C\,\mathrm{e}^{-\tfrac12},\quad C = \mathrm{e}^{C_1}
		\end{align*}
		
		\begin{equation}
			x^2(1+z) = C 
			\label{eq:exact63}
		\end{equation}
		Substituting \(y = 1\) in~\eqref{eq:exact62} we get
		\begin{equation}
			\phi(x) = 1 + z 
			\label{eq:exact64}
		\end{equation}
		From~\eqref{eq:exact63}:
		\[
		z = \frac{C}{x^2}\,\mathrm{e}^{-\tfrac{x}{2}} - 1
		\]
		Substituting this into~\eqref{eq:exact64} gives
		\[
		\phi(x) = \frac{C}{x^2}\,\mathrm{e}^{-\tfrac{x}{2}}
		\]
		Hence equation~\eqref{eq:exact62} becomes
		\begin{align*}
			\frac{y+z}{y^{-\tfrac32}} &= \frac{C}{x^2}\,\mathrm{e}^{-\tfrac{x}{2}},\\
			x^2 \,\frac{y+z}{y^{-\tfrac32}\,\mathrm{e}^{-\tfrac{x}{2} }} &= C.
		\end{align*}
		This is the required first integrals of the given PfDE.
		
		% ============  7 ======================%
		\subsection{$ yz dx + (x^2y - zx) dy + (x^2z - xy) dz = 0$}
		\setcounter{equation}{0}
		We are given the Pfaffian differential equation:
		\begin{equation}
			yz dx + (x^2y - zx) dy + (x^2z - xy) dz = 0
			\label{eq:exact71}
		\end{equation}
		
		Let us define the vector \(\vec{X}=(P,Q,R)\), where:
		\[
		P = yz,\quad Q = x^2y - zx,\quad R = x^2z - xy
		\]
		
		Let us check integrability by computing \(\vec{X}\cdot(\nabla\times\vec{X})\):
		\[
		\vec{X}\cdot(\nabla\times\vec{X})
		= 
		\begin{vmatrix}
			P & Q & R \\
			\frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\
			P & Q & R
		\end{vmatrix}
		=
		\begin{vmatrix}
			yz & x^2y - xz & x^2z - xy \\
			\frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\
			yz & x^2y - xz & x^2z - xy
		\end{vmatrix}
		\]
		
		\begin{align*}
			&= yz\bigl((-x)-(-x)\bigr)
			- (x^2y - xz)\bigl((2xz - y)-y\bigr)
			+ (x^2z - xy)\bigl((2xy - z)-z\bigr)\\
			&= yz\cdot0\;-\;(x^2y - xz)\,2(xz-y)\;+\;(x^2z - xy)\,2(xy-z)\\
			&= 2\Bigl[-(x^2y - xz)(xz-y)+(x^2z - xy)(xy-z)\Bigr]\\
			&= 0
		\end{align*}
		
		Since \(\vec{X}\cdot(\nabla\times\vec{X})=0\), so the given PfDE is integrable.\\
	
		\text{Set $ y = \text{constant} \Rightarrow dy = 0 $}\\
		\text{Then the original equation simplifies to}
		\begin{align*}
		yz\,dx + (x^2 z - xy)\,dz = 0\\
		\frac{z}{x} dx + \left( \frac{xz}{y} - 1 \right) dz = 0\\
		\frac{z}{x} dx = \left(1 - \frac{xz}{y} \right) dz
		\end{align*}
		
		So,
		\[
		\frac{dx}{dz} = \frac{x}{z} - \frac{x^2}{y}
		\quad \text{(Bernoulli equation in } x \text{)}
		\]
		
		Let \( u = \frac{1}{x} \), so
		\[
		\frac{du}{dz} = -\frac{1}{x^2} \frac{dx}{dz}
		\Rightarrow -\frac{1}{u^2} \frac{du}{dz} = \frac{1}{uz} - \frac{1}{y}
		\]
		
		Multiplying both sides by \( -u^2 \):
		\[
		\frac{du}{dz} + \frac{u}{z} = \frac{1}{y}
		\quad \text{(Linear ODE)}
		\]
		
		Integrating factor:
		\[
		\mu = e^{\int \frac{1}{z} dz} = z
		\]
		
		Multiply through by \( z \):
		\[
		z \frac{du}{dz} + u = \frac{z}{y}
		\Rightarrow \frac{d}{dz}(uz) = \frac{z}{y}
		\]
		
		Integrate:
		\[
		uz = \frac{1}{y} \int z\,dz = \frac{z^2}{2y} + \phi(y), \text{$\phi(y)$ is function of y since y is held constant }
		\]
		
		Substitute \( u = \frac{1}{x} \):
		\begin{equation}
		\phi(x) = \frac{z}{x} - \frac{z^2}{2y} 
		\label{eq: exact72}
		\end{equation}
		
		\text{Again Set \( z = 1 \Rightarrow dz = 0 \)}
		
		The equation becomes:
		\begin{align*}
		y\,dx + (x^2 y - x)\,dy = 0\\
		dx + (x^2 - \frac{x}{y})\,dy = 0\\
		\frac{dx}{dy} = -x^2 + \frac{x}{y}
		\quad \text{(Bernoulli equation in } x \text{)}
		\end{align*}
		
		Let \( v = \frac{1}{x} \), so
		\[
		\frac{dv}{dy} = -\frac{1}{x^2} \frac{dx}{dy}
		\Rightarrow -\frac{1}{v^2} \frac{dv}{dy} = -v^{-2} + \frac{1}{vy}
		\]
		
		Multiply both sides by \( -v^2 \):
		\[
		\frac{dv}{dy} + \frac{v}{y} = 1
		\quad \text{(Linear ODE)}
		\]
		
		Integrating factor:
		\[
		\mu = e^{\int \frac{1}{y} dy} = y
		\]
		
		Multiply above ODE by y:
		\begin{align*}
		 y \frac{dv}{dy}  + v = y\\
		 \frac{d}{dy} (vy) = y
		\end{align*}
		
		Integrate:
		\[
		 {v}{y} = \frac{y^2}{2} + C, \text{ where C is an arbitrary constant}
		\]
		\text{Substituting $v = \frac{1}{x}$}
		\begin{equation}
			\frac{y}{x} - \frac{y^2}{2} = C
		\end{equation}
		
		\text{Using z = 1 in equation (2) we get }\\
		\begin{equation}
			\phi(x) = \frac{1}{x} - \frac{1}{2y}
		\end{equation}	
		\text{From equation (3) $\frac{1}{x} = \frac{y}{2} + \frac{C}{y}$ and using this value in equation (4)}
		\begin{align*}
			\phi(y)  = (\frac{y}{2} +\frac{C}{y}) - \frac{1}{2y}\\
			\phi(x) = \frac{y}{2} + \frac{2C-1}{2y}
		\end{align*}
		\text{Hence Equation ~\eqref{eq: exact72} becomes}
		\begin{align*}
			&\frac{z}{x} - \frac{z^2}{2y}  = \frac{y}{2} + \frac{2C-1}{2y}\\
			&\frac{2yz}{x} - z^2 = y^2 + 2C -1\\
			&\frac{2yz}{x}  = y^2 + z^2 + 2C -1\\
			&2xyz = x(y^2+z^2+C_1) \text{ ,where C1= 2C-1}\\
			&x(y^2+z^2) - 2yz = -C_1x\\
			&y^2+z^2 - \frac{2yz}{x} = C2 \text{ ,where C2 = -C1}
		\end{align*}
		\text{This is the require primitives of given PfDE}
		
		
		% ============  8 ======================%
		\subsection{$ 2yz dx - 2xz dy - (x^2-y^2)(z-1) dz = 0$}
		\setcounter{equation}{0}
		We are given the Pfaffian differential equation:
		\begin{equation}
			2yz\,dx - 2xz\,dy - (x^2 - y^2)(z - 1)\,dz = 0
			\label{eq:exact82}
		\end{equation}
		
		Let us define the vector \(\vec{X} = (P, Q, R)\), where:
		\[
		P = 2yz,\quad Q = -2xz,\quad R = -(x^2 - y^2)(z - 1)
		\]
		
		Let us check integrability by computing \(\vec{X} \cdot (\nabla \times \vec{X})\):
		\[
		\vec{X} \cdot (\nabla \times \vec{X})
		= 
		\begin{vmatrix}
			P & Q & R \\
			\frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\
			P & Q & R
		\end{vmatrix}
		=
		\begin{vmatrix}
			2yz & -2xz & -(x^2 - y^2)(z - 1) \\
			\frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\
			2yz & -2xz & -(x^2 - y^2)(z - 1)
		\end{vmatrix}
		\]
		
		\begin{align*}
			&= 2yz \left( \frac{\partial}{\partial y} R - \frac{\partial}{\partial z} Q \right)
			- (-2xz) \left( \frac{\partial}{\partial x} R - \frac{\partial}{\partial z} P \right)
			+ \left[-(x^2 - y^2)(z - 1)\right] \left( \frac{\partial}{\partial x} Q - \frac{\partial}{\partial y} P \right) \\
			&= 2yz \left[ 2y(z - 1) - (-2x) \right]
			+ 2xz \left[ 2x(z - 1) - 2z \right]
			- (x^2 - y^2)(z - 1)\left[ -2z - 2z \right] \\
			&= 2yz(2y(z - 1) + 2x)
			+ 2xz(2x(z - 1) - 2z)
			+ 4z(x^2 - y^2)(z - 1) \\
			& = [4xyz + 4y^2 z(z - 1)] + [-4xyz - 4x^2 z(z - 1)] + [4z(x^2 - y^2)(z - 1)]\\
			& = 4y^2 z(z - 1) - 4x^2 z(z - 1) + 4z(x^2 - y^2)(z - 1)
			& = 4z(z - 1)\left[y^2 - x^2 + (x^2 - y^2)\right] \\
			&  = 0
		\end{align*}
		
		Since \(\vec{X} \cdot (\nabla \times \vec{X}) = 0\), the given PfDE is integrable.
		\begin{align*}
			&\text{Divide the original Pfaffian by }x^2+y^2:\\
			&\quad
			\frac{2yz}{x^2+y^2}\,dx
			\;-\;
			\frac{2xz}{x^2+y^2}\,dy
			\;-\;
			\frac{(x^2 - y^2)(z-1)}{x^2+y^2}\,dz
			\;=\;0.\\[1em]
		\end{align*}
		\text{Lets introduce the cylindrical coordinates $x = r\cos\theta,\;y = r\sin\theta,\;x^2+y^2=r^2$}\\
		\text{Then}\\
		\begin{align*}
			&x^2 + y^2 = r^2\\
			&x^2 - y^2 = r^2 \cos(2\theta)\\
			&dx = \cos(\theta)\,dr - r\sin(\theta)\,d\theta\\
			&dy = \sin(\theta)\,dr + r\cos(\theta)\,d\theta
		\end{align*}
		
		\text{Substituting these values, we get}
		\begin{align*}
			& \frac{2z\left(r\sin(\theta)(\cos(\theta)\,dr - r\sin(\theta)\,d\theta) - r\cos(\theta)(\sin(\theta)\,dr + r \cos(\theta)\,d\theta)\right)}{r^2} \\
			&{} - \frac{r^2\cos(2\theta) (z-1)}{r^2}\,dz = 0\\
			& \frac{2zr\left( \sin\theta \cos\theta\,dr - r\sin^2\theta\,d\theta - \sin\theta \cos\theta\,dr - r \cos^2\theta\,d\theta \right)}{r^2} - \cos(2\theta)(z-1)\,dz = 0\\
			& \frac{2zr\left(-r(\sin^2\theta + \cos^2\theta)\,d\theta \right)}{r^2} - \cos(2\theta)(z-1)\,dz = 0\\
			& -2z\,d\theta - \cos(2\theta)(z-1)\,dz = 0\\
			& -2z d\theta = \cos2\theta(z-1)dz\\
			&\text{Separating variables}\\
			& \frac{d\theta}{\cos2\theta} = -\frac{z-1}{2z} dz\\
			& \text{Integrating both sides:} \\
			& \int \sec2\theta d\theta = -\frac{1}{2} \int \frac{z-1}{z} dz + C \quad \text{where $C$ is an arbitrary constant}\\
			& \frac{1}{2} \ln(\sec2\theta + \tan2\theta) = -\frac{1}{2} \int \big(1-\frac{1}{z}\big) dz +C\\
			& \ln(\sec2\theta + \tan2\theta) =  z-\ln(z) +C\\
			& \text{Substituting back the Cartesian coordinates:} \\
			& \sec2\theta = \frac{1}{\cos2\theta} = \frac{r^2}{x^2+y^2} \text{  and } \tan2\theta = \frac{2\sin\theta \cos\theta}{\cos^2\theta - \sin^2\theta}\\   
			& \text{Hence, } \sec2\theta + \tan2\theta = \frac{r^2+2xy}{x^2-y^2} = \frac{x^2+2xy + y^2}{x^2-y^2} = \frac{(x+y)^2}{x^2-y^2}\\
			& \text{So the solution becomes} \\
			& \ln\big(\frac{(x+y)^2}{x^2-y^2}) = z- \ln(z) +C\\
			& \text{Exponentiating both sides}\\
			& \frac{(x+y)^2}{x^2-y^2} = z e^{-z} e^C \\
			& \boxed{\frac{x+y}{z(x-y)}  = e^{-z} K \text{ where K = $e^C$}}\\
			& \text{This is the required primitive.}
		\end{align*}
